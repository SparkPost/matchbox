### Grid API

<Prop
  name="start"
  type="string"
> 
  xs, sm, md, lg, xl <br/>
  Aligns to the left at specified breakpoint
</Prop>

<Prop
  name="center"
  type="string"
> 
  xs, sm, md, lg, xl <br/>
  Aligns to the center at specified breakpoint
</Prop>

<Prop
  name="end"
  type="string"
> 
  xs, sm, md, lg, xl <br/>
  Aligns to the Right at specified breakpoint
</Prop>

<Prop
  name="top"
  type="string"
> 
  xs, sm, md, lg, xl <br/>
  Aligns to the top at specified breakpoint
</Prop>

<Prop
  name="middle"
  type="string"
> 
  xs, sm, md, lg, xl <br/>
  Aligns to the middle at specified breakpoint
</Prop>

<Prop
  name="bottom"
  type="string"
> 
  xs, sm, md, lg, xl <br/>
  Aligns to the bottom at specified breakpoint
</Prop>

<Prop
  name="around"
  type="string"
> 
  xs, sm, md, lg, xl <br/>
  Sets space around columns at specified breakpoint
</Prop>

<Prop
  name="between"
  type="string"
> 
  xs, sm, md, lg, xl
  Sets space between columns at specified breakpoint
</Prop>

<Prop
  name="children"
  type="node"
> 
  The contents of the Grid
</Prop>

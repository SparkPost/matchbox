# Tooltip

Use this component to explain the function or action of an interactive element. They are revealed through hover and focus.

---

### Usage

<LiveCode>
  <LiveCode.Content 
    disableResize
    title="Default Tooltip" 
    code={`
      <Tooltip content="I am an explanation">
        <Button outline>Hover Me</Button>
      </Tooltip>
    `}
  />
  <LiveCode.Content 
    disableResize
    title="Light Tooltip" 
    code={`
      <Tooltip bg="white" color="gray.900" border="400" content="I am an explanation">
        <Button outline>Hover Me</Button>
      </Tooltip>
    `}
  />
  
</LiveCode>

### Toggle API
<Prop
  name="as"
  type="string"
> 
  Element nape to render the tooltip activator as
</Prop>
<Prop
  name="id"
  type="string"
  required
> 
  The ID of the tooltip container
</Prop>
<Prop
  name="content"
  type="React node"
> 
  The content of the tooltip
</Prop>
<Prop
  name="disabled"
  type="bool"
  required
> 
  Prevents the tooltip from activating
</Prop>
<Prop
  name="active"
  type="bool"
> 
  To be removed, not used
</Prop>
<Prop
  name="left"
  type="bool"
> 
  Causes the tooltip to render right aligned with the trigger so the content overflows left	
</Prop>
<Prop
  name="right"
  type="bool"
  defaultValue="true"
> 
  Causes the tooltip to render left aligned with the trigger so the content overflows right	
</Prop>
<Prop
  name="top"
  type="bool"
> 
  Causes the tooltip to render above the trigger	
</Prop>
<Prop
  name="bottom"
  type="bool"
  defaultValue="true"
> 
  Causes the tooltip to render below the trigger	
</Prop>
<Prop
  name="horizontalOffset"
  type="string"
  defaultValue="0px"
> 
  Horizontally offset the tooltip content	
</Prop>
<Prop
  name="forcePosition"
  type="string"
  defaultValue="false"
> 
  Disables automatic positioning	
</Prop>
<Prop
  name="children"
  type="React node"
> 
  The children of the tooltip	
</Prop>
<Prop
  name="portalId"
  type="string"
> 
  Element ID for the portal that will house tooltips. Appends to body if not provided.	
</Prop>

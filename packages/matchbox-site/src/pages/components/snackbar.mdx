# Snackbar

This component is used for real-time confirmation or feedback that a user-initiated action has taken place.

---

### Usage

<LiveCode>
  <LiveCode.Content
    title="Default Informational"
    code={`
      <Snackbar>
        Snackbar Content
      </Snackbar>
    `}
  />
  <LiveCode.Content
    title="Success"
    code={`<Snackbar status="success" onDismiss={() => alert('Clicked')}>
  Snackbar Content
</Snackbar>`}
  />
  <LiveCode.Content
    title="Warning"
    code={`
<Snackbar status="warning" onDismiss={() => alert('Clicked')}>

### Snackbar API

<Prop name="status" type="string">
  <div>One of <inlineCode>default</inlineCode> <inlineCode>success</inlineCode> <inlineCode>error</inlineCode> <inlineCode>warning</inlineCode></div>
  <div>The status of the banner, determines the color scheme</div>
</Prop>
<Prop name="maxWidth" type="number">
  The maximum width of the snackbar
</Prop>
<Prop name="onDismiss" type="function">
  Callback for when the Snackbar is dismissed	
</Prop>
<Prop name="children" type="React node">
  Content to be displayed inside the banner	
</Prop>

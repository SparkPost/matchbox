### Usage

  <LiveCode.Content
    title="Basic"
    code={`
      <Checkbox id="id" label="Check Me" />
    `}
  />
  <LiveCode.Content

      <Checkbox id="id" label="Check Me" labelHidden />
    `}
  />
  <LiveCode.Content
    title="with JSX in Label"
    code={`
      <Checkbox

          </span>
        }
      />
    `}
  />
  <LiveCode.Content
    title="Disabled"
    code={`
      <>
        <Checkbox id="id5" label="Disabled" disabled />
        <Checkbox id="id6" label="Disabled Checked" disabled checked />
      </>
    `}
  />
  <LiveCode.Content
    title="With Help Text"
    code={`
      <Checkbox id="id" label="Check Me" helpText="Check this box" />
    `}
  />
  <LiveCode.Content
    title="With Error and Required"
    code={`
      <Checkbox id="id" label="Check Me" error="I'm an error" required />
    `}

  <LiveCode.Content
    title="Group with Label"
    code={`
      <div>
        <Checkbox.Group label="Example">

      <div>

        </Checkbox.Group>

  />

### Checkbox API

<Prop
  name="id"
  type="string"
  required
> 
  The ID for the form input
</Prop>

<Prop
  name="checked"
  type="bool"
> 
  Whether the input is selected or not
</Prop>

<Prop
  name="label"
  type="node"
> 
  The label for the Checkbox
</Prop>

<Prop
  name="labelHidden"
  type="bool"
> 
  Hides the label visually, but still accessible to screen readers
</Prop>

<Prop
  name="disabled"
  type="bool"
> 
  Whether the input is disabled or not
</Prop>

<Prop
  name="required"
  type="bool"
> 
  Whether the input is required or not
</Prop>

<Prop
  name="value"
  type="string"
> 
  Value for the input
</Prop>

<Prop
  name="error"
  type="string"
> 
  Renders an error message message below the input
</Prop>

<Prop
  name="onChange"
  type="function"
  defaultValue="(e) => void"
> 
  Callback when input checked state changes
</Prop>

<Prop
  name="onFocus"
  type="function"
  defaultValue="(e) => void"
> 
  Callback when input becomes focused
</Prop>

<Prop
  name="onBlur"
  type="function"
  defaultValue="(e) => void"
> 
  Callback when input loses focus
</Prop>

<Prop
  name="helpText"
  type="node"
> 
  Additional text to help the user that renders below the input
</Prop>

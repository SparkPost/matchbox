### TextField API

<Prop
  name="align"
  type="string"
> 
  left, center, right <br/>
  The text alignment of the input
</Prop>

<Prop
  name="id"
  type="string"
  required
> 
  The ID for the form input
</Prop>

<Prop
  name="multiline"
  type="bool"
> 
  Renders a multiline textarea
</Prop>

<Prop
  name="type"
  type="string"
  defaultValue="text"
> 
  text, email, number, password, search, tel, url, date <br/>
  The type of the input
</Prop>

<Prop
  name="placeholder"
  type="string"
> 
  Placeholder text for the input
</Prop>

<Prop
  name="autofocus"
  type="bool"
> 
  Whether the input will be automatically focused on initial render or not
</Prop>

<Prop
  name="disabled"
  type="bool"
> 
  Whether the input is disabled or not
</Prop>

<Prop
  name="readOnly"
  type="bool"
> 
  Whether the value of the input is able to be changed or not
</Prop>

<Prop
  name="required"
  type="bool"
> 
  Whether the input is required or not
</Prop>

<Prop
  name="resize"
  type="string"
> 
  none, both, horizontal, vertical, block, inline <br/>
  A value to determine how a multiline input can be resized
</Prop>

<Prop
  name="label"
  type="string"
> 
  Label for the TextField
</Prop>

<Prop
  name="labelHidden"
  type="bool"
> 
  Hides the label visually, but still accessible to screen readers
</Prop>

<Prop
  name="helpText"
  type="string"
> 
  Additional text to help the user that renders below the input
</Prop>

<Prop
  name="connectLeft"
  type="node"
> 
  Content rendered inline to the left of the input
</Prop>

<Prop
  name="connectRight"
  type="node"
> 
  Content rendered inline to the right of the input
</Prop>

<Prop
  name="prefix"
  type="node"
> 
  Content rendered before the input value (i.e. currency, icons, etc.)
</Prop>

<Prop
  name="suffix"
  type="node"
> 
  Content rendered after the input value (i.e. icons)
</Prop>

<Prop
  name="error"
  type="string"
> 
  Renders an error message below the input
</Prop>

<Prop
  name="errorInLabel"
  type="bool"
> 
  Optionally renders the error message to the right of the input's label
</Prop>

<Prop
  name="onChange"
  type="function"
  defaultValue="(e) => void"
> 
  Callback when input value state changes
</Prop>

<Prop
  name="onFocus"
  type="function"
  defaultValue="(e) => void"
> 
  Callback when input is focused
</Prop>

<Prop
  name="onBlur"
  type="function"
  defaultValue="(e) => void"
> 
  Callback when input loses focus
</Prop>

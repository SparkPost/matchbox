### Page API

<Prop name="title" type="number">
  Main title of the page
</Prop>
<Prop name="subtitle" type="number">
  Subtitle that appears below the title
</Prop>
<Prop name="primaryAction" type="number">
  Main call to action button - accepts button props
</Prop>
<Prop name="primaryArea" type="React node">
  Alternative to `primaryAction`. Renders content in place of main call to
  action button
</Prop>
<Prop name="secondaryActions" type="array">
  Secondary actions that appear in an actionlist if more than one item and as a
  button beside primary action if only one item
</Prop>
<Prop name="breadcrumbAction" type="object">
  The back link action
</Prop>
<Prop name="empty" type="object">
  Optional empty state object
</Prop>
<Prop name="children" type="React node">
  Contents of the page
</Prop>

### Modal API

<Prop name="open" type="bool" required>
  Controlled open state of the modal. Provided by the useModal hook.
</Prop>
<Prop name="onClose" type="function">
  Callback that is called on key down 'Escape' and on click outside modal
  content. Provided by the useModal hook.
</Prop>
<Prop name="children" type="node">
  The content of the modal overlay
</Prop>
<Prop name="closeOnEscape" type="bool" defaultProps="true">
  Whether or not the modal should close when the escape key is pressed
</Prop>
<Prop name="closeOnOutsideClick" type="bool" defaultProps="true">
  Whether or not the modal should close when clicking outside the modal
</Prop>
<Prop name="maxWidth" type="number" defaultValue="1200">
  The maximum width of the modal content
</Prop>
<Prop name="portalId" type="string">
  Controls the target container ID for the rendering React portal
</Prop>

### Modal.Header System Props

None

### Modal.Header API

<Prop name="children" type="node">
  The content of the modal header
</Prop>
<Prop name="showCloseButton" type="bool" defaultProps="true">
  Whether or not to show the close button
</Prop>
<Prop name="onClose" type="function">
  Callback to close the modal when the close button is clicked. Provided by the
  useModal hook.
</Prop>

# Snackbar

This component is used for real-time confirmation or feedback that a user-initiated action has taken place.

---

### Usage

<LiveCode>
  <LiveCode.Content
    title="Default Informational"
    code={`
      <Snackbar>
        Snackbar Content
      </Snackbar>
    `}
  />
  <LiveCode.Content
    title="Success"
    code={`<Snackbar status="success" onDismiss={() => alert('Clicked')}>
  Snackbar Content
</Snackbar>`}
  />
  <LiveCode.Content
    title="Warning"
    code={`
<Snackbar status="warning" onDismiss={() => alert('Clicked')}>
  Snackbar Content
</Snackbar>
    `}
  />
  <LiveCode.Content
    title="Error"
    code={`
<Snackbar status="error" onDismiss={() => alert('Clicked')}>
  Snackbar Content
</Snackbar>
    `}
  />
  <LiveCode.Content
    title="Long Content"
    code={`
<Snackbar maxWidth={500}>
  This one is large enough to get into some bacon ipsum dolor amet pork loin tri-tip turkey capicola. Rump doner short ribs biltong burgdoggen meatloaf. Prosciutto pork loin bacon, biltong landjaeger salami ham spare ribs flank cupim porchetta leberkas.
</Snackbar>
    `}
  />
</LiveCode>

### Snackbar API

<Prop
  name="status"
  type="string"
  values={['default', 'success', 'error', 'warning']}
>
  The status of the banner, determines the color scheme
</Prop>
<Prop name="maxWidth" type="number">
  The maximum width of the snackbar
</Prop>
<Prop name="onDismiss" type="function">
  Callback for when the Snackbar is dismissed
</Prop>
<Prop name="children" type="React node">
  Content to be displayed inside the banner
</Prop>

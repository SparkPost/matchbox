# DatePicker

This component lets users choose dates visually through a calendar, and uses <ExternalLink to="http://react-day-picker.js.org/">react-day-picker</ExternalLink> under the hood.

---

### Usage

<LiveCode>
  <LiveCode.Content
    title="Two Month Datepicker"
    code={`function Component() {
      const to = new Date();
      const from = new Date(new Date().setDate(new Date().getDate() - 15));
      const initial = new Date(new Date().setDate(new Date().getDate() - 30));
      const selectedDays = { to, from };
      const modifiers = {
        firstSelected: day => DateUtils.isSameDay(day, selectedDays.from),
        lastSelected: day => DateUtils.isSameDay(day, selectedDays.to),
        inBetween: day => DateUtils.isDayBetween(day, selectedDays.from, selectedDays.to),
      };
      return (
        <DatePicker
          modifiers={modifiers}
          numberOfMonths={2}
          initialMonth={initial}
          disabledDays={{ after: new Date() }}
          toMonth={new Date()}
          selectedDays={selectedDays}
          onDayClick={d => console.log('click', d)}
          m="400"
        />
      );
    }
    `}
  />
</LiveCode>

### Popover API

<Prop
  name="id"
  type="string"
> 
  The popover ID
</Prop>

<Prop
  name="trigger"
  type="node"
> 
  An element used to trigger the popover. Click event is handled automatically if this component is uncontrolled
</Prop>

<Prop
  name="open"
  type="bool"
> 
  Opens the popover. By default, open state is handled automatically. Passing this value in will turn this into a controlled component
</Prop>

<Prop
  name="left"
  type="bool"
> 
  Causes the popover to render left aligned with the trigger so the content overflows left
</Prop>

<Prop
  name="right"
  type="bool"
> 
  Causes the popover to render right aligned with the trigger so the content overflows right
</Prop>

<Prop
  name="top"
  type="bool"
> 
  Causes the popover to render above the trigger
</Prop>

<Prop
  name="bottom"
  type="bool"
> 
  Causes the popover to render below the trigger
</Prop>

<Prop
  name="onClose"
  type="function"
  defaultValue="(e) => void"
> 
  Callback function that is called when clicking outside the popover, or hitting escape
</Prop>

<Prop
  name="children"
  type="node"
> 
  The content of the popover
</Prop>

<Prop
  name="as"
  type="string"
> 
  div, span <br/>
  Whether to render the popover as a div or a span (block or inline)
</Prop>

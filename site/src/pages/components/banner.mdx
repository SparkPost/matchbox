### Banner API

<Prop
  name="status"
  type="string"
  values={['default', 'success', 'warning', 'danger', 'info']}
>
  The status of the banner, determines the color scheme.
</Prop>

<Prop name="title" type="string">
  The title of the banner
</Prop>

<Prop name="onDismiss" type="function" defaultValue="(e) => void">
  Callback for when the banner is dismissed
</Prop>

<Prop name="action" type="object" required>
  Used for single action button
</Prop>

<Prop name="actions" type="array">
  Renders multiple action buttons
</Prop>

<Prop name="children" type="node">
  Content to be displayed inside the banner
</Prop>

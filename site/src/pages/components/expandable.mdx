### Expandable API

<Prop name="defaultOpen" type="bool">
  Default open state when not controlled
</Prop>

<Prop name="icon" type="node">
  Optional React node for an image or icon. Works best with an SVG optimized for
  40x40
</Prop>

<Prop name="id" type="string" required>
  Required for accessibility between header and content
</Prop>

<Prop name="open" type="bool">
  Controls the open state
</Prop>

<Prop name="onToggle" type="function" defaultValue="(e) => void">
  Callback for when expandable is toggled. Optional, but required when
  controlling open state
</Prop>

<Prop name="subtitle" type="node">
  Content area beneath header title
</Prop>

<Prop name="title" type="string">
  The header title
</Prop>

<Prop name="accent" type="bool | string">
  orange, blue, red, yellow, green, purple, navy, gray <br />
  Renders an accent border at the top of the expandable panel. If bool is used it
  defaults to blue
</Prop>

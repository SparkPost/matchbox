### Usage

  <LiveCode.Content
    title="Chevrons"
    code={`
      <CodeBlock code={${codeSnippet}}/>
    `}

  <LiveCode.Content
    title="Numbers"

      <CodeBlock numbered code={${codeSnippet}}/>
    `}

  <LiveCode.Content

    code={`

    `}

### CodeBlock System Props

None

---

### CodeBlock API

<Prop name="code" type="string" required>
    The code to render in the CodeBlock
</Prop>

<Prop name="className" type="string">
    Optional class name for the CodeBlock
</Prop>

<Prop name="numbered" type="bool">
  Render line numbers (defaults to use chevrons)
</Prop>

<Prop name="height" type="number">
  Fixed height in pixels of the CodeBlock
</Prop>

<Prop name="dark" type="bool">
  Whether the code block is styled dark or light (defaults to light)
</Prop>

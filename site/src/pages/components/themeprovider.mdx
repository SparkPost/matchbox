# ThemeProvider

A wrapper component for Matchbox's theming. Injects Matchbox's styling into all Matchbox components beneath it in the component tree, via the context API.

---

### Usage

```jsx
import { ThemeProvider } from '@sparkpost/matchbox';

function App() {
  return <ThemeProvider>Your App Here</ThemeProvider>;
}
```

---

### ThemeProvider API

<Prop name="children" type="node" required>
  Child contents will have access to Matchbox's styling. All Matchbox components
  need to be a child of ThemeProvider.
</Prop>

<Prop name="skipGlobalStyles" type="bool">
  Skips injecting global styles. Useful for test runners.
</Prop>

<Prop name="target" type="DOM element">
  Provides an alternate DOM node to inject styled-components stylesheets into.
  See{' '}
  <ExternalLink to="https://styled-components.com/docs/api#stylesheetmanager">
    styled-components.com/docs/api#stylesheetmanager
  </ExternalLink>
  .
</Prop>

<Prop name="theme" type="object">
  Optionally overwrites Matchbox's system props. See{' '}
  <ExternalLink to="https://github.com/SparkPost/matchbox/blob/main/packages/matchbox/src/components/ThemeProvider/theme.js">
    github.com/SparkPost/matchbox/blob/main/packages/matchbox/src/components/ThemeProvider/theme.js
  </ExternalLink>
  .
</Prop>

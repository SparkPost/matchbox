# UnstyledLink

---

### Usage

<LiveCode>
  <LiveCode.Content
    title="External link"
    code={`
      <UnstyledLink to="https://sparkpost.com" external>
        SparkPost
      </UnstyledLink>
    `}
  />
  <LiveCode.Content
    title="With an on click handler"
    code={`
      <UnstyledLink onClick={() => alert('Click!')}>A link</UnstyledLink>
    `}
  />
</LiveCode>

---

### System Props

None

---

### Page API

<Prop name="children" type="React node">
  Contents of the link
</Prop>
<Prop name="disabled" type="bool">
  Disables the link
</Prop>
<Prop name="component" type="elementType">
  Optional component wrapper for the link
</Prop>
<Prop name="external" type="bool">
  Whether the link goes to an external url or not
</Prop>
<Prop name="to" type="string | object">
  The url that the link will direct to
</Prop>
<Prop name="title" type="string">
  The title of the link
</Prop>

# Layout

Use the Layout component to quickly generate common page layouts for the SparkPost app.

---

### Usage

<LiveCode>
  <LiveCode.Content
    title="Annotated Layout"
    code={`
      <Layout>
        <Layout.Section annotated>
          <Layout.SectionTitle>Section Title</Layout.SectionTitle>
          <Text color="gray.700" fontSize="200">Annotated Section Text</Text>
        </Layout.Section>
        <Layout.Section>
          <Panel>
            <Panel.Section>
              Panel Content
            </Panel.Section>
            <Panel.Section>
              Panel Content
            </Panel.Section>
          </Panel>
        </Layout.Section>
      </Layout>`}
  />
  <LiveCode.Content
    title="One and two column layouts"
    code={`
    <>
      <Layout>
        <Layout.Section>
          <Box p="400" bg="yellow.200">One Column</Box>
        </Layout.Section>
      </Layout>
      <Layout>
        <Layout.Section>
          <Box p="400" bg="yellow.200">Two Column</Box>
        </Layout.Section>
        <Layout.Section>
          <Box p="400" bg="yellow.200">Two Column</Box>
        </Layout.Section>
      </Layout>
    </>
    `}
  />
</LiveCode>

### Layout API

<Prop
  name="collapseBelow"
  type="string"
  defaultValue='"md"'
  values={['"xs"', '"sm"', '"md"', '"lg"', '"xl"']}
>
  Child columns vertically stack below this breakpoint
</Prop>
<Prop name="children" type="node">
  The content of the Layout
</Prop>

### Layout.Section API

<Prop name="annotated" type="bool">
  Automatically adds correct width of an annotated column
</Prop>
<Prop name="children" type="node">
  The content of the Layout Section
</Prop>

### Layout.SectionTitle API

<Prop name="as" type="elementType" defaultValue='"h2"'>
  DOM element type to render as
</Prop>

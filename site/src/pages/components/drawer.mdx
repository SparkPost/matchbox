### Drawer API

<Prop name="children" type="React node">
  Drawer children, only use Drawer namespaced components
</Prop>
<Prop name="closeOnEscape" type="bool" defaultValue="true">
  If true, closes the drawer when the escape key is pressed
</Prop>
<Prop name="closeOnOutsideClick" type="bool" defaultValue="true">
  If true, closes the drawer when the clicking outside the drawer
</Prop>
<Prop name="id" type="string" required>
  Dom element ID of the drawer, provided by <inlineCode>useDrawer</inlineCode>
</Prop>
<Prop name="onChange" type="func">
  Callback function when drawer open state changes
</Prop>
<Prop name="onClose" type="func">
  Callback function to close the drawer, provided by{' '}
  <inlineCode>useDrawer</inlineCode>
</Prop>
<Prop name="open" type="bool" required>
  Open state of the drawer, provided by <inlineCode>useDrawer</inlineCode>
</Prop>
<Prop name="portalId" type="string">
  Controls the target container ID for the rendering React portal
</Prop>
<Prop
  name="position"
  type="string"
  defaultValue="right"
  values={['"right"', '"left"']}
>
  Position of the drawer
</Prop>
<Prop name="portalId" type="string">
  Controls the target container ID for the rendering React portal
</Prop>

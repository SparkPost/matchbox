### Button API
<Prop
  name="color"
  type="string"
> 
  gray, blue, red <br /> The color of the button
</Prop>

<Prop
  name="disabled"
  type="bool"
>
  Whether the button is disabled or not
</Prop>

<Prop
  name="destructive"
  type="bool"
>
  Whether the button is used for destructive action or not (i.e. delete) - Will be the destructive red color
</Prop>

<Prop
  name="flat"
  type="bool"
>
  The button will have a transparent background and the text color will correspond with the color prop
</Prop>

<Prop
  name="outline"
  type="bool"
>
  The button will have a gray border and text corresponding with the color prop
</Prop>

<Prop
  name="outlineBorder"
  type="bool"
>
  The button will have a border matching the color of text corresponding with the color prop
</Prop>

<Prop
  name="size"
  type="string"
  defaultValue="default"
>
  small, large, default <br />
  The size of the button
</Prop>

<Prop
  name="fullWidth"
  type="bool"
>
  Whether the button is the full width of its parent
</Prop>

<Prop
  name="submit"
  type="bool"
>
  Whether the button is a submit button or regular button
</Prop>

<Prop
  name="to"
  type="string | object"
>
  The location the button will route to
</Prop>

<Prop
  name="title"
  type="string"
>
  The title of the button
</Prop>

<Prop
  name="external"
  type="bool"
>
  Whether the button takes you to an external link or not
</Prop>

<Prop
  name="component"
  type="element type"
>
  Component to render the button as. (i.e. 'span')
</Prop>

<Prop
  name="children"
  type="react node"
>
  Children of the button
</Prop>

### Button API

<Prop name="color" type="string" values={['gray', 'blue', 'red']}>
  The color of the button
</Prop>

<Prop name="children" type="react node">
  Children of the button
</Prop>

<Prop name="as" type="elementType">
  Component to render the button as. (i.e. 'span')
</Prop>

<Prop name="disabled" type="bool">
  Whether the button is disabled or not
</Prop>

<Prop name="destructive" type="bool" deprecated>
  Use the color prop instead
</Prop>

<Prop name="external" type="bool">
  Whether the button takes you to an external link or not
</Prop>

<Prop name="flat" type="bool" deprecated>
  Use the variant prop instead
</Prop>

<Prop name="fullWidth" type="bool">
  Whether the button is the full width of its parent
</Prop>

<Prop name="outline" type="bool" deprecated>
  Use the variant prop instead
</Prop>

<Prop name="outlineBorder" type="bool" deprecated>
  Use the variant prop instead
</Prop>

<Prop
  name="size"
  type="string"
  defaultValue="default"
  values={['small', 'large', 'default']}
>
  The size of the button
</Prop>

<Prop name="submit" type="bool">
  Whether the button is a submit button or regular button
</Prop>

<Prop name="to" type="string | object">
  The location the button will route to
</Prop>

<Prop name="title" type="string">
  The title of the button
</Prop>

<Prop
  name="variant"
  type="string"
  defaultValue='"filled"'
  values={['"filled"', '"outline"', '"mutedOutline"', '"text"']}
>
  Button style variant
</Prop>

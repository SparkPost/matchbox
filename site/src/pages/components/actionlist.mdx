### ActionList API

<Prop name="actions" type="[{content: ReactNode, ...}]" deprecated>
  Actions e.g. [&#123; content: 'action label', onClick: callback() &#125;]
  Note: each item can include an optional "section" key that will be used to
  auto group into sections, declaratively
</Prop>
<Prop name="children" type="node">
  Content of the ActionList
</Prop>
<Prop name="sections" type="[[{content: ReactNode, ...}]]" deprecated>
  Creates sections e.g. [[&#123; content: 'action label', onClick: callback()
  &#125;]]
</Prop>
<Prop name="maxHeight" type="string | number">
  Max height of list
</Prop>
<Prop name="groupByKey" type="string" defaultvalue="'section'" deprecated>
  Group by key used to auto group actions into sections, defaults to "section"
</Prop>

### ActionList.Action API

<Prop name="children" type="node">
  Content of the ActionList Action
</Prop>

<Prop name="helpText" type="string">
  Renders a string of secondary text
</Prop>

<Prop name="highlighted" type="bool">
  Highlights the action. Useful for wrappers that manage focus or keyboard
  navigation, such as downshift
</Prop>

<Prop
  name="is"
  type="string"
  values={['"link"', '"button"']}
  defaultValue='"link"'
>
  If set to "button", renders the action as a button in the DOM, otherwise,
  renders an acnhor element
</Prop>

<Prop name="selected" type="bool" deprecated>
  Use <Link to="/components/checkbox">the checkbox component</Link> inside a
  popover instead
</Prop>

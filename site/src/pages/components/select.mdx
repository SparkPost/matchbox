### Select API

<Prop name="id" type="string">
  The ID for the form input
</Prop>

<Prop name="options" type="array" required>
  Array of options with &#123; value, label &#125;, Strings, or Numbers
</Prop>

<Prop name="disabled" type="bool">
  Whether the input is disabled or not
</Prop>

<Prop name="required" type="bool">
  Whether the input is required or not
</Prop>

<Prop name="label" type="string">
  Label for the input
</Prop>

<Prop name="labelHidden" type="bool">
  Hides the label visually, but still accessible to screen readers
</Prop>

<Prop name="helpText" type="node">
  Addition text to help the user that renders below the input
</Prop>

<Prop name="error" type="string">
  Renders an error message below the input
</Prop>

<Prop name="errorInLabel" type="bool">
  Optionally renders the error message to the right of the input's label
</Prop>

<Prop name="onChange" type="function" defaultValue="(e) => void">
  Callback when input value state changes
</Prop>

<Prop name="onFocus" type="function" defaultValue="(e) => void">
  Callback when input becomes focused
</Prop>

<Prop name="onBlur" type="function" defaultValue="(e) => void">
  Callback when input loses focus
</Prop>

### Columns API

<Prop name="align" type="string" values={['"left"', '"right"', '"center"']}>
  Horizontal alignment of the columns
</Prop>
<Prop name="alignY" type="string" values={['"top"', '"bottom"', '"center"']}>
  Vertical alignment of the columns
</Prop>
<Prop
  name="collpaseBelow"
  type="string"
  values={['"xs"', '"sm"', '"md"', '"lg"', '"xl"']}
>
  Child columns vertically stack below this breakpoint
</Prop>
<Prop name="children" type="node">
  The content of the columns â€“ Use the `Column` component only
</Prop>
<Prop name="reverse" type="bool">
  Reverses the order of the columns
</Prop>
<Prop name="space" type="string" defaultValue="500">
  Gutter between columns
</Prop>

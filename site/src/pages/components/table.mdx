### Table API

<Prop name="data" type="array">
  List of data to be displayed in the table. Used to auto generate table from
  data
</Prop>

<Prop name="children" type="node">
  The content of the table
</Prop>

### Table.HeaderCell API

<Prop name="value" type="node">
  The value of the cell
</Prop>

<Prop name="className" type="string">
  Optional class name for the cell
</Prop>

### Table.SortButton API

<Prop
  name="direction"
  type="string"
  values={['"asc"', '"desc"', 'null', 'undefined']}
>
  Sets the direction of the button's icon
</Prop>

<Prop name="children" type="node">
  The content of the button
</Prop>

---

### Table.Row & Table.TotalsRow API

<Prop name="rowData" type="array">
  The data for the table row
</Prop>

<Prop name="className" type="string">
  Optional class name for the row
</Prop>

<Prop name="children" type="node">
  The content of the row
</Prop>

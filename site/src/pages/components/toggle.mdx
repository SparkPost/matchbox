### Toggle API

<Prop name="checked" type="bool | string">
  Whether the input is checked or not
</Prop>
<Prop name="defaultChecked" type="bool | string">
  Whether the input is checked by default or not
</Prop>
<Prop name="disabled" type="bool">
  Whether the input is disabled or not
</Prop>
<Prop name="id" type="string" required>
  The ID for the form input
</Prop>
<Prop name="label" type="string">
  Label for the toggle
</Prop>
<Prop name="onChange" type="function" defaultValue="(event) => void">
  Callback when input checked state changes
</Prop>
<Prop name="onFocus" type="function" defaultValue="(event) => void">
  Callback when input becomes focused
</Prop>
<Prop name="onBlur" type="function" defaultValue="(event) => void">
  Callback when input loses focus
</Prop>
<Prop name="required" type="bool">
  Whether the input is required or not
</Prop>
<Prop name="value" type="bool | string">
  Value for the form input
</Prop>

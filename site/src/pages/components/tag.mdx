### Tabs API

<Prop name="color" type="string">
  <div>
    blue | yellow | red | purple | green | magenta | teal | lightGray | darkGray
  </div>
  <div>Sets the Tag color</div>
</Prop>
<Prop name="onRemove" type="function">
  Renders a remove button and is called when the button is clicked
</Prop>
<Prop name="children" type="React node">
  Tag content
</Prop>

### Video API

<Prop name="autoPlay" type="bool" defaultValue="true">
  Autoplays the video
</Prop>

<Prop name="children" type="node">
  Contents of the video element
</Prop>

<Prop name="controls" type="bool" defaultValue="false">
  Displays controls on the video
</Prop>

<Prop name="loop" type="bool" defaultValue="false">
  Auto-loops the video
</Prop>

<Prop name="muted" type="bool" defaultValue="true">
  Mutes the video
</Prop>

<Prop name="role" type="string">
  The role attribute of the video element
</Prop>

### Video.Source API

<Prop name="className" type="string">
  Optional class name for the video
</Prop>

<Prop name="source" type="string" required>
  URL source of the video
</Prop>

<Prop name="type" type="string">
  Format type of the video source file
</Prop>

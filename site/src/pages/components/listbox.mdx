### ListBox API

<Prop name="id" type="string">
  The ID for the listbox
</Prop>

<Prop name="value" type="string | number">
  The value of the listbox
</Prop>

<Prop name="defaultValue" type="string | number">
  The default value of the listbox
</Prop>

<Prop name="disabled" type="bool">
  Whether the listbox is disabled or not
</Prop>

<Prop name="required" type="bool">
  Whether the listbox is required or not
</Prop>

<Prop name="label" type="string">
  Label for the listbox
</Prop>

<Prop name="labelHidden" type="bool">
  Whether the label is hidden or not
</Prop>

<Prop name="helpText" type="node">
  Addition text to help the user that renders below the listbox
</Prop>

<Prop name="error" type="string">
  Renders an error message below the listbox
</Prop>

<Prop name="errorInLabel" type="bool">
  Optionally renders the error message to the right of the listbox's label
</Prop>

<Prop name="onChange" type="function" defaultValue="(e) => void">
  Callback when listbox value state changes
</Prop>

<Prop name="onFocus" type="function" defaultValue="(e) => void">
  Callback when listbox becomes focused
</Prop>

<Prop name="onBlur" type="function" defaultValue="(e) => void">
  Callback when listbox loses focus
</Prop>
